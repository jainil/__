Custom event delegation without using jquery is fairly easy: 
          function delegate(container, event, elDescriptor, callback) {
            container.addEventListener(event, function (e) {
              if (e.target && e.target.matches(elDescriptor)) {
                callback(e.target);
              }
            })
          }

          var container = document.querySelector('.container');

          delegate(container, 'click', 'div.box', function (el) {
            alert(`Box ${el.innerHTML} clicked`);
          })
